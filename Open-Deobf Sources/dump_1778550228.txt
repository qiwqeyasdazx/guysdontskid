local TranslationData = {
    Languages = {
        ["English"] = "en",
        ["Indonesian"] = "id",
        ["Spanish"] = "es",
        ["French"] = "fr",
        ["German"] = "de",
        ["Japanese"] = "ja",
        ["Korean"] = "ko",
        ["Vietnamese"] = "vi",
        ["Thai"] = "th",
        ["Russian"] = "ru",
        ["Portuguese"] = "pt"
    },
    RegisteredElements = {}, -- [instance] = {original = "..."}
    Cache = {}
}

local function TranslateText(text, target)
    if text == "" or text == nil then return "" end
    local cacheKey = target .. ":" .. text
    if TranslationData.Cache[cacheKey] then return TranslationData.Cache[cacheKey] end

    local success, result = pcall(function()
        local url = "https://translate.googleapis.com/translate_a/single?client=gtx&sl=auto&tl="..target.."&dt=t&q="..game:GetService("HttpService"):UrlEncode(text)
        local response = game:HttpGet(url)
        local decoded = game:GetService("HttpService"):JSONDecode(response)
        local result = ""
        if decoded and decoded[1] then
            for _, v in ipairs(decoded[1]) do
                if v[1] then result = result .. v[1] end
            end
        end
        return result
    end)
    
    if success and result and result ~= "" then
        TranslationData.Cache[cacheKey] = result
        return result
    end
    return text
end

local function UpdateAllTranslations()
    local target = TranslationData.Languages[SettingsLib.TranslateLanguage] or "en"
    for instance, data in pairs(TranslationData.RegisteredElements) do
        if not instance or not instance.Parent then 
            TranslationData.RegisteredElements[instance] = nil
        else
            if SettingsLib.AutoTranslate then
                task.spawn(function()
                    local translated = TranslateText(data.original, target)
                    data.isTranslating = true
                    instance.Text = translated
                    data.isTranslating = false
                end)
            else
                instance.Text = data.original
            end
        end
    end
end

local function RegisterForTranslation(instance)
    if not instance:IsA("TextLabel") and not instance:IsA("TextButton") and not instance:IsA("TextBox") then return end
    
    local data = { original = instance.Text, isTranslating = false }
    TranslationData.RegisteredElements[instance] = data
    
    instance:GetPropertyChangedSignal("Text"):Connect(function()
        if data.isTranslating then return end
        data.original = instance.Text
        if SettingsLib.AutoTranslate then
            task.spawn(function()
                local target = TranslationData.Languages[SettingsLib.TranslateLanguage] or "en"
                local translated = TranslateText(data.original, target)
                data.isTranslating = true
                instance.Text = translated
                data.isTranslating = false
            end)
        end
    end)

    if SettingsLib.AutoTranslate and instance.Text ~= "" then
        task.spawn(function()
            local target = TranslationData.Languages[SettingsLib.TranslateLanguage] or "en"
            local translated = TranslateText(data.original, target)
            data.isTranslating = true
            instance.Text = translated
            data.isTranslating = false
        end)
    end
end

local function Tween(instance, properties, duration, style, direction)
    duration = duration or 0.3
    style = style or Enum.EasingStyle.Quart
    direction = direction or Enum.EasingDirection.Out
    local tweenInfo = TweenInfo.new(duration, style, direction)
    local tween = TweenService:Create(instance, tweenInfo, properties)
    tween:Play()
    return tween
end

local function ApplyGradient(obj)
    local UIGradient = Instance.new("UIGradient")
    UIGradient.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Theme.Accent1),
        ColorSequenceKeypoint.new(1, Theme.Accent2)
    }
    UIGradient.Rotation = 45
    UIGradient.Parent = obj
    return UIGradient
end

local function ApplyStroke(obj, thickness, transparency)
    local UIStroke = Instance.new("UIStroke")
    UIStroke.Thickness = thickness or 0.8
    UIStroke.Transparency = transparency or 0.4
    UIStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    UIStroke.Color = Color3.new(1,1,1)
    UIStroke.Parent = obj
    ApplyGradient(UIStroke)
    return UIStroke
end

local function Create(className, properties, children)
	local instance = Instance.new(className)
	for k, v in pairs(properties or {}) do
		instance[k] = v
	end
    
    if (className:find("Text") or instance:IsA("TextLabel") or instance:IsA("TextButton") or instance:IsA("TextBox")) then
        RegisterForTranslation(instance)
    end

    if children then
        for _, child in ipairs(children) do
            child.Parent = instance
        end
    end
	return instance
end

local function CreateRounded(Parent, Size)

    })
end

local function MakeDraggable(topbarobject, object)
	local Dragging, DragInput, DragStart, StartPosition
	
	local function updateDrag(input)
		local Delta = input.Position - DragStart
		Tween(object, {
            Position = UDim2.new(StartPosition.X.Scale, StartPosition.X.Offset + Delta.X, StartPosition.Y.Scale, StartPosition.Y.Offset + Delta.Y)
        }, 0.15)
	end
	
	topbarobject.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			Dragging = true
			DragStart = input.Position
			StartPosition = object.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then Dragging = false end
			end)
		end
	end)
	
	topbarobject.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			DragInput = input
		end
	end)
	
	UserInputService.InputChanged:Connect(function(input)
		if input == DragInput and Dragging then updateDrag(input) end
	end)
end

local function MakeResizable(dragHandle, mainObj, outlineObj, minSize, maxSize)
	local Dragging, DragInput, DragStart, StartSize
	
	local function updateResize(input)
		local Delta = input.Position - DragStart
		local newWidth = math.clamp(StartSize.X.Offset + Delta.X, minSize.X, maxSize.X)
		local newHeight = math.clamp(StartSize.Y.Offset + Delta.Y, minSize.Y, maxSize.Y)
		
		Tween(outlineObj, {Size = UDim2.new(0, newWidth, 0, newHeight)}, 0.1)
		Tween(mainObj, {Size = UDim2.new(0, newWidth, 0, newHeight)}, 0.1)
	end
	
	dragHandle.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			Dragging = true
			DragStart = input.Position
			StartSize = outlineObj.Size
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then Dragging = false end
			end)
		end
	end)
	
	dragHandle.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			DragInput = input
		end
	end)
	
	UserInputService.InputChanged:Connect(function(input)
		if input == DragInput and Dragging then updateResize(input) end
	end)
end

    local desc = isConfig and Config.Desc or Config
    local icon = isConfig and Config.Icon or (icon or "rbxassetid://102268449481061")

    local NotifShadow = Create("Frame", {
        Name = "NotifShadow",
        Parent = NotificationFrame,
        BackgroundColor3 = Theme.Background,
        AnchorPoint = Vector2.new(1, 1),
        Position = UDim2.new(1, 400, 1, -20),
        Size = UDim2.new(0, 320, 0, 65),
        ClipsDescendants = true
    }, {

    NotifShadow.BackgroundTransparency = 0.05

    local NotifIcon = Create("ImageLabel", {
        Parent = NotifShadow,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 14, 0.5, -4),
        AnchorPoint = Vector2.new(0, 0.5),
        Size = UDim2.new(0, 24, 0, 24),
        Image = icon or "rbxassetid://136911603207160",
        ImageColor3 = Theme.TextWhite
    })

    Create("TextLabel", {
        Parent = NotifShadow,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 48, 0, 12),
        Size = UDim2.new(1, -80, 0, 20),
        Font = Enum.Font.GothamMedium,
        Text = title,
        TextColor3 = Theme.TextWhite,
        TextSize = 15,
        TextXAlignment = Enum.TextXAlignment.Left
    })

    Create("TextLabel", {
        Parent = NotifShadow,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 48, 0, 32),
        Size = UDim2.new(1, -60, 0, 20),
        Font = Enum.Font.GothamMedium,
        Text = desc,
        TextColor3 = Theme.TextGrey,
        TextSize = 13,
        TextXAlignment = Enum.TextXAlignment.Left,
        TextWrapped = true
    })

    local CloseButton = Create("ImageButton", {
        Parent = NotifShadow,
        BackgroundTransparency = 1,
        Position = UDim2.new(1, -26, 0, 10),
        Size = UDim2.new(0, 16, 0, 16),
        Image = "rbxassetid://10747384394",
        ImageColor3 = Theme.TextGrey
    })
    
    local ProgressBarBg = Create("Frame", {
        Parent = NotifShadow,
        BackgroundColor3 = Theme.Section,
        BackgroundTransparency = 0.5,
        Position = UDim2.new(0, 4, 1, -4),
        Size = UDim2.new(1, -8, 0, 4),
        BorderSizePixel = 0
    }, { Create("UICorner", { CornerRadius = UDim.new(0, 4) }) })

    local ProgressBar = Create("Frame", {
        Parent = ProgressBarBg,
        BackgroundColor3 = Color3.new(1,1,1),
        Position = UDim2.new(0, 0, 0, 0),
        Size = UDim2.new(1, 0, 1, 0),
        BorderSizePixel = 0
    }, { Create("UICorner", { CornerRadius = UDim.new(0, 4) }) })
    ApplyGradient(ProgressBar)

    local notifData = { Frame = NotifShadow }
    
    -- Function to close this specific notification
    local isClosing = false
    local function closeNotif()
        if isClosing then return end
        isClosing = true
        
        -- Remove from table
        for i, v in ipairs(NotificationList) do
            if v == notifData then
                table.remove(NotificationList, i)
                break
            end
        end
        UpdateNotifications()

        local tween = Tween(NotifShadow, {Position = UDim2.new(1, 400, NotifShadow.Position.Y.Scale, NotifShadow.Position.Y.Offset)}, 0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
        tween.Completed:Connect(function() NotifShadow:Destroy() end)
    end

    -- Insert at the beginning so newest is at Index 1 (Bottom)
    table.insert(NotificationList, 1, notifData)

    -- Force pop the 5th (Oldest) notification
    if #NotificationList > 4 then
        local oldestNotif = table.remove(NotificationList, 5)
        Tween(oldestNotif.Frame, {Position = UDim2.new(1, 400, oldestNotif.Frame.Position.Y.Scale, oldestNotif.Frame.Position.Y.Offset)}, 0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out).Completed:Connect(function()
            oldestNotif.Frame:Destroy()
        end)
    end

    UpdateNotifications()

    -- Interactions
    CloseButton.MouseEnter:Connect(function() Tween(CloseButton, {ImageColor3 = Theme.Error}, 0.2) end)
    CloseButton.MouseLeave:Connect(function() Tween(CloseButton, {ImageColor3 = Theme.TextGrey}, 0.2) end)
    CloseButton.MouseButton1Click:Connect(closeNotif)

    -- Progress Bar Auto Close
    local progressTween = Tween(ProgressBar, {Size = UDim2.new(0, 0, 1, 0)}, NotifDuration, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut)
    progressTween.Completed:Connect(closeNotif)
end

        function main:Section(Config)
            local isConfig = type(Config) == "table"
            local title = isConfig and (Config.Title or "Section") or (Config or "Section")
            local desc = isConfig and Config.Desc or nil
            local icon = GetIcon(isConfig and Config.Icon or nil)
            local startOpen = isConfig and (Config.Opened == true) or false
            ElementOrderCount = ElementOrderCount + 1
            table.insert(RelzUILib.ElementQueue, "Building Section: " .. title)

            local headerHeight = desc and 52 or 40
            local isOpen = startOpen
            local SectionElementOrder = 0

            -- Outer wrapper: holds header + clipped content
            local SectionWrapper = Create("Frame", {
                Parent = MainFramePage,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, headerHeight),
                ClipsDescendants = false,
                LayoutOrder = ElementOrderCount
            })

            -- Header button (clickable)
            local SectionHeader = Create("TextButton", {
                Parent = SectionWrapper,
                BackgroundColor3 = Theme.Section,
                BackgroundTransparency = 0.5,
                Size = UDim2.new(1, 0, 0, headerHeight),
                Text = "",
                AutoButtonColor = false,
                ZIndex = 2
            }, { Create("UICorner", { CornerRadius = UDim.new(0, 10) }) })

            -- Hover glow on header

            SectionHeader.MouseButton1Down:Connect(function() Tween(SectionHeader, {BackgroundTransparency = 0.3}, 0.1) end)
            SectionHeader.MouseButton1Up:Connect(function() Tween(SectionHeader, {BackgroundTransparency = 0.5}, 0.2) end)

            -- Section icon (optional)
            local textLeftOffset = 14
            if icon and tostring(icon) ~= "" and tostring(icon) ~= "rbxassetid://" then
                local SectionIcon = Create("ImageLabel", {
                    Parent = SectionHeader,
                    BackgroundTransparency = 1,
                    Position = UDim2.new(0, 14, 0.5, 0),
                    AnchorPoint = Vector2.new(0, 0.5),
                    Size = UDim2.new(0, 18, 0, 18),
                    Image = icon,
                    ImageColor3 = Color3.new(1,1,1),
                    ScaleType = Enum.ScaleType.Fit,
                    ZIndex = 5
                })
                ApplyGradient(SectionIcon)
                textLeftOffset = 40
            end

            -- Title text container inside header
            local HeaderTextContainer = Create("Frame", {
                Parent = SectionHeader,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, textLeftOffset, 0.5, 0),
                AnchorPoint = Vector2.new(0, 0.5),
                Size = UDim2.new(1, -(textLeftOffset + 38), 0, 0),
                AutomaticSize = Enum.AutomaticSize.Y,
                ZIndex = 3
            }, { Create("UIListLayout", { SortOrder = Enum.SortOrder.LayoutOrder, Padding = UDim.new(0, 3) }) })

            Create("TextLabel", {
                Parent = HeaderTextContainer,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, 0),
                AutomaticSize = Enum.AutomaticSize.Y,
                Font = Enum.Font.GothamMedium,
                Text = title,
                TextColor3 = Theme.TextWhite,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
                TextYAlignment = Enum.TextYAlignment.Top,
                TextWrapped = true,
                LayoutOrder = 1,
                ZIndex = 3
            })

            if desc then
                Create("TextLabel", {
                    Parent = HeaderTextContainer,
                    BackgroundTransparency = 1,
                    Size = UDim2.new(1, 0, 0, 0),
                    AutomaticSize = Enum.AutomaticSize.Y,
                    Font = Enum.Font.GothamMedium,
                    Text = desc,
                    TextColor3 = Theme.TextDark,
                    TextSize = 11,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextYAlignment = Enum.TextYAlignment.Top,
                    TextWrapped = true,
                    LayoutOrder = 2,
                    ZIndex = 3
                })
            end

            -- Chevron icon (rotates when expanded)
            local ChevronIcon = Create("ImageLabel", {
                Parent = SectionHeader,
                BackgroundTransparency = 1,
                AnchorPoint = Vector2.new(1, 0.5),
                Position = UDim2.new(1, -14, 0.5, 0),
                Size = UDim2.new(0, 16, 0, 16),
                Image = "rbxassetid://10709790948",
                ImageColor3 = Color3.new(1,1,1),
                Rotation = startOpen and 180 or 0,
                ZIndex = 3
            })
            ApplyGradient(ChevronIcon)

            -- Content area (shown when open)
            local SectionContent = Create("Frame", {
                Parent = SectionWrapper,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 0, 0, headerHeight + 4),
                Size = UDim2.new(1, 0, 0, 0),
                AutomaticSize = Enum.AutomaticSize.Y,
                ClipsDescendants = false,
                Visible = startOpen
            }, {
                Create("UIListLayout", { SortOrder = Enum.SortOrder.LayoutOrder, Padding = UDim.new(0, 6) }),
                Create("UIPadding", { PaddingLeft = UDim.new(0, 0), PaddingTop = UDim.new(0, 2), PaddingBottom = UDim.new(0, 2) })
            })

            -- Section icon (optional)

            -- Helper to recalculate SectionWrapper height
            local function updateWrapperHeight()
                if isOpen then
                    local contentH = SectionContent.UIListLayout.AbsoluteContentSize.Y
                    local pad = 10
                    local total = headerHeight + 4 + contentH + pad
                    SectionWrapper.Size = UDim2.new(1, 0, 0, total)
                else
                    SectionWrapper.Size = UDim2.new(1, 0, 0, headerHeight)
                end
            end

            -- Listen for content size changes
            SectionContent.UIListLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                local contentH = SectionContent.UIListLayout.AbsoluteContentSize.Y
                if isOpen then
                    updateWrapperHeight()
                end
            end)

            -- Open / Close functions
            local function openSection()
                isOpen = true
                SectionContent.Visible = true
                if contentBg then contentBg.Visible = true end
                Tween(ChevronIcon, {Rotation = 180, ImageColor3 = Theme.TextWhite}, 0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
                task.defer(function()
                    updateWrapperHeight()
                end)
            end

            local function closeSection()
                isOpen = false
                Tween(ChevronIcon, {Rotation = 0, ImageColor3 = Theme.TextGrey}, 0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
                SectionWrapper.Size = UDim2.new(1, 0, 0, headerHeight)
                task.defer(function()
                    SectionContent.Visible = false
                    if contentBg then contentBg.Visible = false end
                end)
            end

            if startOpen then
                task.defer(function() openSection() end)
            end

            SectionHeader.MouseButton1Click:Connect(function()
                if isOpen then closeSection() else openSection() end
            end)

            -- Build section API using proxy pattern:
            -- Each method calls the IDENTICAL main function, then reparents the result to SectionContent.
            -- This guarantees Section elements are 100% identical to Tab elements.
            local section = {}

            local function proxyElement(callFn)
                -- Snapshot current children of MainFramePage
                local beforeCount = #MainFramePage:GetChildren()
                -- Run the actual element creation (same code as Tab)
                local result = callFn()
                -- Find and move all newly created children to SectionContent
                local children = MainFramePage:GetChildren()
                for i = #children, beforeCount + 1, -1 do
                    local child = children[i]
                    SectionElementOrder = SectionElementOrder + 1
                    child.LayoutOrder = SectionElementOrder
                    child.Parent = SectionContent
                end
                -- Undo the ElementOrderCount increment(s) made by main's method
                ElementOrderCount = ElementOrderCount - (math.max(0, #MainFramePage:GetChildren() + (#children - beforeCount) - beforeCount))
                return result
            end

            function section:Button(cfg, d, cb)
                return proxyElement(function() return main:Button(cfg, d, cb) end)
            end

            function section:Toggle(cfg, c, d, cb)
                return proxyElement(function() return main:Toggle(cfg, c, d, cb) end)
            end

            function section:Dropdown(cfg, opt, d, var, cb)
                return proxyElement(function() return main:Dropdown(cfg, opt, d, var, cb) end)
            end

            function section:Slider(cfg, min2, max2, def, cb)
                return proxyElement(function() return main:Slider(cfg, min2, max2, def, cb) end)
            end

            function section:Textbox(cfg, dis, d, cb)
                return proxyElement(function() return main:Textbox(cfg, dis, d, cb) end)
            end

            function section:Paragraph(cfg, d)
                return proxyElement(function() return main:Paragraph(cfg, d) end)
            end

            function section:Line()
                return proxyElement(function() return main:Line() end)
            end

            function section:Separator(cfg, txt)
                return proxyElement(function() return main:Separator(cfg, txt) end)
            end

            function section:DiscordInfo(cfg)
                return proxyElement(function() return main:DiscordInfo(cfg) end)
            end

            function section:Open() openSection() end
            function section:Close() closeSection() end

            return section
        end

		return main
	end
